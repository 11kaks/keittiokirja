
\begin{recipe}
[% 
    preparationtime = {\unit[45]{min}},
    bakingtime={\unit[10]{min}},
    portion = {\portion{2}},
    %calory={\unit[3]{kJ}},
    %source = {Somebody you used know}
]
{Thaiahven}
\label{rcp:tahiahven}
    
    \graph
    {% pictures
        small=pic/thaiahven2,     % small picture
        big=pic/thaiahven1  % big picture
    }
    \introduction{%
    	Täysin päästä temmmottu, mutta yllättävän thaikkuvibainen kalaruoka kahdelle.
    }
    
    
    \ingredients{%
	\unit[2 \unitfrac{3}{4}]{dl}& vettä \\
				& vehnäjauhoja\\
    2 & kokonaista ahventa \\
    2 & pientä sipulia\\
    2 & chiliä\\
    \unit[3]{kynttä} & valksipulia \\
    \unit[1]{rkl} & kalakastiketta \\
    \unit[1]{rkl} & osterikastiketta \\
    \unit[1]{rkl} & tumma soijakastiketta \\
    pari rkl & rypsiölyjä \\
    }
    
    \preparation{%
   		Fileoi affenet. Jos et osaa, opettele jostain. Itse leikkasin filetillä ranteeseen: 
   		älä tee niin. Paloittele fileet puikkosyötävän kokoisiksi könteiksi. Laita kalapalat 
   		keskikokoseen kuppiin.
   		
   		Kuori ja pilko vihannekset, ja laita ne kalojen seuraksi kuppiin, ettei kaloilla 
   		olisi niin yksinäistä. 
   		
   		Kaada samaan kuppiin kaikki kastikkeet. Jos ei kaikkea löydy, niin ei haittaa. Saattaa tulla 
   		muutenkin liian suolasta. Kastikkeet antaa safkaan iloisen thaikkuviban. Tumma soija
   		tekee safkasta tummaa--jos et tykkää siitä ni käytä vaaleeta. 
   		
   		Jätä mössö maustumaan ja keitä ootellessa puolisen desiä riisiä.
   		
   		Kun riisi on valmista, laita pannu levylle ja levy melkein täysille. Paista kulhon sisältö 
   		kuumalla pannulla reilussa öljyssä. Kymmenisen minuuttia riittää sanoisin.
   		
   		Tarjoile vaikkapa kurkku- ja miniluumukirsikkatomaattiviipaleiden kanssa.
    }
    
\end{recipe}
